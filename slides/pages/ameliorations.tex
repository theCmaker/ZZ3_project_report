	\framesubtitle{Concept}
	\begin{block}{Fonctionnement}
		\begin{itemize}
			\item Automatisation des mouvements dans un algorithme
			\item Pas d'amélioration $\Rightarrow$ passer au mouvement suivant

	\end{block}
